f[n][m]=max\{f[n-1][m-k]+pmf[n][k],0<k<m\} 
$$
$$
f[n][m]=\begin{cases}
pmf[1][m] & n=1 \\
max\{f[n-1][m-k]+pmf[n][k]\} & n>1,0<k<m
    
\end{cases}

$$

$$
f[i][j]=\begin{cases}
    f[i][j]=0 & i=j \\
    D[i][j]+min{f[i][k]+f[k][j]}&i\leq k<j
\end{cases}
$$
